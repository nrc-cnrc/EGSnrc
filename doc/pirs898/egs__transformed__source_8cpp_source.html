40 EGS_TransformedSource::EGS_TransformedSource(EGS_Input *input,
41  EGS_ObjectFactory *f) : EGS_BaseSource(input,f), source(0), T(0) {
42  EGS_Input *isource = input->takeInputItem("source",false);
43  if (isource) {
44  source = EGS_BaseSource::createSource(isource);
45  delete isource;
46  }
47  if (!source) {
48  string sname;
49  int err = input->getInput("source name",sname);
50  if (err)
51  egsWarning("EGS_TransformedSource: missing/wrong inline source "
52  "definition and missing wrong 'source name' input\n");
53  else {
54  source = EGS_BaseSource::getSource(sname);
55  if (!source) egsWarning("EGS_TransformedSource: a source named %s"
56  " does not exist\n");
57  }
58  }
59  EGS_AffineTransform *t = EGS_AffineTransform::getTransformation(input);
60  setUp(t);
61  delete t;
62 }

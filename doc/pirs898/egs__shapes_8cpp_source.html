118  }
119  else {
120  egsWarning("EGS_BoxShape::createShape: invalid 'box size' input\n");
121  result = 0;
122  }
123  if (t) {
124  delete t;
125  }
126  result->setName(input);
127  return result;

155 EGS_Object *EGS_CylinderShape::createObject(EGS_Input *input) {
156  EGS_Float r, H;
157  int err = input->getInput("radius",r);
158  if (err) {
159  egsWarning("EGS_CylinderShape::getShape: wrong/missing 'radius'"
160  " input\n");
161  return 0;
162  }
163  err = input->getInput("height",H);
164  if (err) {
165  egsWarning("EGS_CylinderShape::getShape: wrong/missing 'height'"
166  " input\n");
167  return 0;
168  }
169  vector<EGS_Float> phi_range;
170  bool set_phi = false;
171  if (!input->getInput("phi range",phi_range) && phi_range.size() == 2) {
172  set_phi = true;
173  phi_range[0] *= M_PI/180;
174  phi_range[1] *= M_PI/180;
175  }
176  EGS_AffineTransform *t = EGS_AffineTransform::getTransformation(input);
177  if (t) {
178  EGS_CylinderShape *result = new EGS_CylinderShape(r,H,t);
179  result->setName(input);
180  if (set_phi) {
181  result->setPhiRange(phi_range[0],phi_range[1]);
182  }
183  delete t;
184  return result;
185  }
186  vector<EGS_Float> Xo, A;
187  int err1 = input->getInput("midpoint",Xo);
188  int err2 = input->getInput("axis",A);
189  bool has_Xo = (err1 == 0 && Xo.size() == 3);
190  bool has_A = (err2 == 0 && A.size() == 3);
191  EGS_CylinderShape *result;
192  if (has_Xo && has_A) result = new EGS_CylinderShape(r,H,
193  EGS_Vector(Xo[0],Xo[1],Xo[2]),EGS_Vector(A[0],A[1],A[2]));
194  else if (has_Xo) result = new EGS_CylinderShape(r,H,
195  EGS_Vector(Xo[0],Xo[1],Xo[2]));
196  else if (has_A) result = new EGS_CylinderShape(r,H,
197  EGS_Vector(0,0,0),EGS_Vector(A[0],A[1],A[2]));
198  else {
199  result = new EGS_CylinderShape(r,H);
200  }
201  result->setName(input);
202  if (set_phi) {
203  result->setPhiRange(phi_range[0],phi_range[1]);
204  }
205  return result;
206 }

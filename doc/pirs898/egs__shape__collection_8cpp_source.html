42 EGS_ShapeCollection::EGS_ShapeCollection(const vector<EGS_BaseShape *> &Shapes,
43  const vector<EGS_Float> &Probs, const string &Name, EGS_ObjectFactory *f) :
44  EGS_BaseShape(Name,f), nshape(0) {
45  int n1 = Shapes.size(), n2 = Probs.size();
46  if (n1 < 2 || n2 < 2 || n1 != n2) {
47  egsWarning("EGS_ShapeCollection::EGS_ShapeCollection: invalid input\n");
48  return;
49  }
50  nshape = n1;
51  shapes = new EGS_BaseShape* [nshape];
52  EGS_Float *p = new EGS_Float [nshape];
53  for (int j=0; j<nshape; j++) {
54  shapes[j] = Shapes[j];
55  shapes[j]->ref();
56  p[j] = Probs[j];
57  }
58  table = new EGS_SimpleAliasTable(nshape,p);
59  delete [] p;
60 }

64  EGS_SHAPE_COLLECTION_EXPORT EGS_BaseShape *createShape(EGS_Input *input,
65  EGS_ObjectFactory *f) {
66  if (!input) {
67  egsWarning("createShape(shape collection): null input?\n");
68  return 0;
69  }
70  vector<EGS_BaseShape *> shapes;
71  vector<EGS_Float> probs;
72  EGS_Input *ishape;
73  while ((ishape = input->takeInputItem("shape",false))) {
74  EGS_BaseShape *shape = EGS_BaseShape::createShape(ishape);
75  if (!shape) {
76  egsWarning("createShape(shape collection): got null shape\n");
77  }
78  else {
79  shapes.push_back(shape);
80  }
81  delete ishape;
82  }
83  vector<string> snames;
84  int err = input->getInput("shape names",snames);
85  if (!err && snames.size() > 0) {
86  for (unsigned int j=0; j<snames.size(); j++) {
87  EGS_BaseShape *shape = EGS_BaseShape::getShape(snames[j]);
88  if (!shape) egsWarning("createShape(shape collection): no shape "
89  "named %s exists\n",snames[j].c_str());
90  else {
91  shapes.push_back(shape);
92  }
93  }
94  }
95  err = input->getInput("probabilities",probs);
96  bool ok = true;
97  if (err) {
98  egsWarning("createShape(shape collection): no 'probabilities' input\n");
99  ok = false;
100  }
101  if (shapes.size() < 2) {
102  egsWarning("createShape(shape collection): at least 2 shapes are "
103  "needed for a shape collection, you defined %d\n",shapes.size());
104  ok = false;
105  }
106  if (shapes.size() != probs.size()) {
107  egsWarning("createShape(shape collection): the number of shapes (%d)"
108  " is not the same as the number of input probabilities (%d)\n");
109  ok = false;
110  }
111  for (unsigned int i=0; i<probs.size(); i++) {
112  if (probs[i] < 0) {
113  egsWarning("createShape(shape collection): probabilities must not"
114  " be negative, but your input probability %d is %g\n",
115  i,probs[i]);
116  ok = false;
117  }
118  }
119  if (!ok) {
120  for (unsigned int j=0; j<shapes.size(); j++) {
121  EGS_Object::deleteObject(shapes[j]);
122  }
123  return 0;
124  }
125  EGS_ShapeCollection *s = new EGS_ShapeCollection(shapes,probs,"",f);
126  s->setName(input);
127  s->setTransformation(input);
128  return s;
129  }

317  }
318  }

320  int err = input->getInput(name,*str_v);
321  if (!err) {
322  vector<string> str = *str_v;

332  }
333  }
334  *str_v = str;
335  }
336  }
337  else if (type == 5) {
338  int err = input->getInput(name,*f_v);
339  if (!err) {

345  number=nitem;

348  }
349  }

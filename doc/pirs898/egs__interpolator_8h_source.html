151  inline EGS_Float interpolate(EGS_Float x) const {
152  if (x > xmin && x < xmax) {
153  int i = (int)(ax + bx*x);
154  if (i < 0) {
155  i=0;
156  }
157  return a[i] + b[i]*x;
158  }
159  else if (x <= xmin) {
160  return fmin;
161  }
162  else {
163  return fmax;
164  }
165  };
